Install CLI
Install once, log in, then use the hosted service directly.
For Agents
# Read
{{ .InstallDocURL }}

# Then run
curl -fsSL {{ .InstallScriptURL }} | sh

# After install
llm-wiki system info --base-url {{ .InstallBaseURL }}
llm-wiki auth login --base-url {{ .InstallBaseURL }}
llm-wiki auth whoami --base-url {{ .InstallBaseURL }}

MCP
Install MCP
Use the remote MCP endpoint or the local stdio bridge with a bearer token.
For Agents
# Remote MCP endpoint
{{ .MCPURL }}

# Auth header
Authorization: Bearer <llm-wiki-token>

# Local stdio bridge
LLM_WIKI_TOKEN=<llm-wiki-token> npx --prefix npm/llm-wiki-mcp llm-wiki-mcp --base-url {{ .InstallBaseURL }}
For Humans
npm install --prefix npm/llm-wiki-mcp --package-lock=false
LLM_WIKI_TOKEN=<llm-wiki-token> npx --prefix npm/llm-wiki-mcp llm-wiki-mcp --base-url {{ .InstallBaseURL }}
